# i_part
# it will return the weight vector and the input to this function are X,Y,lamda which are in numpy formate
def mylinridgereg(X, Y, lamda):
    X_transpose=np.transpose(X)
    res = np.dot(X_transpose,X)
    b = np.identity(res.shape[0], dtype = float)
    b=b*lamda
    w=res+b
    try:
        inverse = np.linalg.inv(w)
    except np.linalg.LinAlgError:
        pass
        print('matrix is not invertible')
    weight1=np.dot(inverse,X_transpose)
    final_w=np.dot(weight1,Y)
    return final_w
# ii_part
# it will return the y_pred values and the input to this function are X,weights which are in numpy formate
def mylinridgeregeval(X, weights):
    res=np.dot(X,weights)
    return res
# iii_part
# it will return the mean square error value and the input to this function are y_actaul and y_pred which are in numpy formate
def meansquarederr(A, B):
    mse = (np.square(A - B)).mean()
    return mse
